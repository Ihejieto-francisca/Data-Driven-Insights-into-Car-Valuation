import pandas as pd
import matplotlib.pyplot as plt

df = pd.read_csv("car_price_dataset.csv")
brand_stats = df.groupby('Brand').agg({'Engine_Size': 'mean', 'Price': 'mean'}).reset_index()

plt.figure(figsize=(12, 6))
plt.bar(brand_stats['Brand'], brand_stats['Engine_Size'], label='Avg Engine Size')
plt.bar(brand_stats['Brand'], brand_stats['Price'], alpha=0.5, label='Avg Price')
plt.xlabel('Brand')
plt.ylabel('Value')
plt.title('Engine Size vs. Price by Brand')
plt.xticks(rotation=45)
plt.legend()
plt.show()
